**Goal:** Build a Next.js + FastAPI web UI that lets users browse historical analysis reports and trigger new analyses with real-time progress streaming.

**Architecture:** FastAPI backend scans the `reports/` directory and exposes REST + SSE endpoints; Next.js frontend renders markdown reports in a tabbed viewer and streams analysis progress via SSE. No database, no auth.

**Tech Stack:** FastAPI, uvicorn, Next.js (App Router), TypeScript, Tailwind CSS, react-markdown, shadcn/ui

## Context: How Reports Work

Reports are saved by `cli/main.py:save_report_to_disk()` to:
```
reports/{TICKER}_{YYYYMMDD_HHMMSS}/
  1_analysts/   market.md, sentiment.md, news.md, fundamentals.md
  2_research/   bull.md, bear.md, manager.md
  3_trading/    trader.md
  4_risk/       aggressive.md, conservative.md, neutral.md
  5_portfolio/  decision.md
  complete_report.md
```

Analysis is triggered via `TradingAgentsGraph(config=config).propagate(ticker, date)`.
The graph takes 5-15 minutes to run. Progress is tracked via agent callbacks.

## Task 2: FastAPI – Report Listing Endpoint

**Files:**
- Create: `web/backend/main.py`

**What it does:** Scan `reports/` at the project root, parse `{TICKER}_{YYYYMMDD_HHMMSS}` folder names, return a grouped list.

**Step 1: Write `web/backend/main.py` with the `/api/reports` endpoint**

```python
import os
import re
from pathlib import Path
from fastapi import FastAPI
from fastapi.middleware.cors import CORSMiddleware

# Project root = two levels up from web/backend/
PROJECT_ROOT = Path(__file__).parent.parent.parent
REPORTS_DIR = PROJECT_ROOT / "reports"

app = FastAPI()

app.add_middleware(
    CORSMiddleware,
    allow_origins=["http://localhost:3000"],
    allow_methods=["*"],
    allow_headers=["*"],
)


def parse_report_id(folder_name: str):
    """Parse 'SPY_20260305_155836' into ticker, date, time."""
    match = re.match(r"^([A-Z0-9.]+)_(\d{8})_(\d{6})$", folder_name)
    if not match:
        return None
    ticker, date_str, time_str = match.groups()
    return {
        "id": folder_name,
        "ticker": ticker,
        "date": f"{date_str[:4]}-{date_str[4:6]}-{date_str[6:]}",
        "time": f"{time_str[:2]}:{time_str[2:4]}:{time_str[4:]}",
        "timestamp": f"{date_str}_{time_str}",
    }


@app.get("/api/reports")
def list_reports():
    """List all reports grouped by ticker."""
    if not REPORTS_DIR.exists():
        return {"reports": {}}

    grouped: dict[str, list] = {}
    for folder in sorted(REPORTS_DIR.iterdir(), reverse=True):
        if not folder.is_dir():
            continue
        info = parse_report_id(folder.name)
        if not info:
            continue
        ticker = info["ticker"]
        if ticker not in grouped:
            grouped[ticker] = []
        grouped[ticker].append(info)

    return {"reports": grouped}
```

## Task 3: FastAPI – Report Content Endpoint

**Files:**
- Modify: `web/backend/main.py`

**What it does:** Given a report ID and optional stage/file params, return the markdown content of the file.

**Step 1: Add the report content and structure endpoints to `main.py`**

```python
from fastapi import HTTPException

# Map stage names to directory names
STAGE_DIRS = {
    "analysts": "1_analysts",
    "research": "2_research",
    "trading": "3_trading",
    "risk": "4_risk",
    "portfolio": "5_portfolio",
}

# Files available per stage
STAGE_FILES = {
    "analysts": ["market", "sentiment", "news", "fundamentals"],
    "research": ["bull", "bear", "manager"],
    "trading": ["trader"],
    "risk": ["aggressive", "conservative", "neutral"],
    "portfolio": ["decision"],
}


@app.get("/api/reports/{report_id}/structure")
def get_report_structure(report_id: str):
    """Return which files exist in a report."""
    report_path = REPORTS_DIR / report_id
    if not report_path.exists():
        raise HTTPException(404, "Report not found")

    structure = {"id": report_id, "stages": {}, "has_complete": False}
    structure["has_complete"] = (report_path / "complete_report.md").exists()

    for stage, dir_name in STAGE_DIRS.items():
        stage_path = report_path / dir_name
        if stage_path.exists():
            files = [
                f.stem
                for f in stage_path.iterdir()
                if f.suffix == ".md"
            ]
            structure["stages"][stage] = files

    return structure


@app.get("/api/reports/{report_id}/content")
def get_report_content(report_id: str, stage: str = "complete", file: str = ""):
    """Return markdown content for a specific report file."""
    report_path = REPORTS_DIR / report_id
    if not report_path.exists():
        raise HTTPException(404, "Report not found")

    if stage == "complete":
        md_path = report_path / "complete_report.md"
    else:
        dir_name = STAGE_DIRS.get(stage)
        if not dir_name:
            raise HTTPException(400, f"Unknown stage: {stage}")
        md_path = report_path / dir_name / f"{file}.md"

    if not md_path.exists():
        raise HTTPException(404, "File not found")

    return {"content": md_path.read_text(encoding="utf-8")}
```

## Task 4: FastAPI – Analysis Trigger + SSE Progress

**Files:**
- Modify: `web/backend/main.py`

**What it does:** POST `/api/analyze` starts analysis in a background thread, writing progress messages to an in-memory queue. GET `/api/analyze/stream?job_id=xxx` streams those messages as SSE until done.

**Step 1: Add job management and trigger endpoint**

```python
import uuid
import threading
import asyncio
import sys
from queue import Queue, Empty
from typing import Optional
from pydantic import BaseModel
from sse_starlette.sse import EventSourceResponse

# In-memory job store: job_id -> {"queue": Queue, "status": str, "report_id": str|None}
JOBS: dict[str, dict] = {}


class AnalyzeRequest(BaseModel):
    ticker: str
    date: str  # YYYY-MM-DD format
    llm_provider: str = "openai"
    deep_think_llm: str = "gpt-4o"
    quick_think_llm: str = "gpt-4o-mini"


def run_analysis(job_id: str, ticker: str, date: str, config: dict):
    """Run analysis in background thread, push progress to job queue."""
    q = JOBS[job_id]["queue"]

    def send(msg: str, event: str = "progress"):
        q.put({"event": event, "data": msg})

    try:
        send(f"Starting analysis for {ticker} on {date}...")

        # Add project root to path so tradingagents package is importable
        sys.path.insert(0, str(PROJECT_ROOT))
        from dotenv import load_dotenv
        load_dotenv(PROJECT_ROOT / ".env")
        from tradingagents.graph.trading_graph import TradingAgentsGraph

        send(f"Initializing agents...")
        ta = TradingAgentsGraph(debug=False, config=config)

        send(f"Running analysis pipeline (this takes several minutes)...")
        final_state, decision = ta.propagate(ticker, date)

        # Save report
        from cli.main import save_report_to_disk
        import datetime
        timestamp = datetime.datetime.now().strftime("%Y%m%d_%H%M%S")
        report_id = f"{ticker}_{timestamp}"
        save_path = REPORTS_DIR / report_id
        save_report_to_disk(final_state, ticker, save_path)

        JOBS[job_id]["status"] = "done"
        JOBS[job_id]["report_id"] = report_id
        send(f"Analysis complete! Decision: {decision}", event="done")
        send(report_id, event="report_id")

    except Exception as e:
        JOBS[job_id]["status"] = "error"
        send(f"Error: {str(e)}", event="error")
    finally:
        q.put(None)  # sentinel to signal end of stream


@app.post("/api/analyze")
def start_analysis(req: AnalyzeRequest):
    """Start a new analysis job and return a job_id."""
    from tradingagents.default_config import DEFAULT_CONFIG
    config = DEFAULT_CONFIG.copy()
    config["llm_provider"] = req.llm_provider
    config["deep_think_llm"] = req.deep_think_llm
    config["quick_think_llm"] = req.quick_think_llm

    job_id = str(uuid.uuid4())
    JOBS[job_id] = {"queue": Queue(), "status": "running", "report_id": None}

    thread = threading.Thread(
        target=run_analysis,
        args=(job_id, req.ticker, req.date, config),
        daemon=True,
    )
    thread.start()

    return {"job_id": job_id}


@app.get("/api/analyze/stream")
async def stream_analysis(job_id: str):
    """SSE endpoint to stream analysis progress for a job."""
    if job_id not in JOBS:
        raise HTTPException(404, "Job not found")

    job = JOBS[job_id]
    q = job["queue"]

    async def event_generator():
        loop = asyncio.get_event_loop()
        while True:
            try:
                item = await loop.run_in_executor(None, lambda: q.get(timeout=1))
                if item is None:
                    break
                yield {"event": item["event"], "data": item["data"]}
            except Empty:
                yield {"event": "ping", "data": ""}

    return EventSourceResponse(event_generator())


@app.get("/api/analyze/{job_id}/status")
def get_job_status(job_id: str):
    """Get the current status of an analysis job."""
    if job_id not in JOBS:
        raise HTTPException(404, "Job not found")
    job = JOBS[job_id]
    return {"status": job["status"], "report_id": job.get("report_id")}
```

"use client";
import { useEffect, useState } from "react";
import { listReports, ReportInfo } from "@/lib/api";

interface Props {
  selectedId: string | null;
  onSelect: (id: string) => void;
}

export default function ReportSidebar({ selectedId, onSelect }: Props) {
  const [reports, setReports] = useState<Record<string, ReportInfo[]>>({});
  const [expanded, setExpanded] = useState<Record<string, boolean>>({});

  useEffect(() => {
    listReports().then((data) => {
      setReports(data);
      // Auto-expand all tickers
      const exp: Record<string, boolean> = {};
      Object.keys(data).forEach((t) => (exp[t] = true));
      setExpanded(exp);
    });
  }, []);

  const refresh = () => listReports().then(setReports);

  return (
    <aside className="w-64 min-h-screen bg-gray-50 border-r border-gray-200 p-4 flex flex-col gap-4">
      <h2 className="font-semibold text-gray-700 text-sm uppercase tracking-wide">Reports</h2>

      {Object.keys(reports).length === 0 && (
        <p className="text-sm text-gray-400">No reports yet.</p>
      )}

      {Object.entries(reports).map(([ticker, items]) => (
        <div key={ticker}>
          <button
            className="flex items-center justify-between w-full text-left text-sm font-medium text-gray-800 py-1"
            onClick={() => setExpanded((e) => ({ ...e, [ticker]: !e[ticker] }))}
          >
            <span>{ticker}</span>
            <span className="text-gray-400">{expanded[ticker] ? "▾" : "▸"}</span>
          </button>
          {expanded[ticker] && (
            <ul className="ml-3 mt-1 space-y-1">
              {items.map((r) => (
                <li key={r.id}>
                  <button
                    onClick={() => onSelect(r.id)}
                    className={`w-full text-left text-xs px-2 py-1.5 rounded transition-colors ${
                      selectedId === r.id
                        ? "bg-blue-100 text-blue-800 font-medium"
                        : "text-gray-600 hover:bg-gray-100"
                    }`}
                  >
                    {r.date}
                    <span className="ml-1 text-gray-400">{r.time}</span>
                  </button>
                </li>
              ))}
            </ul>
          )}
        </div>
      ))}

      <button
        onClick={refresh}
        className="mt-auto text-xs text-gray-400 hover:text-gray-600 text-left"
      >
        ↻ Refresh
      </button>
    </aside>
  );
}

"use client";
import { useEffect, useState } from "react";
import { getReportStructure, getReportContent, ReportStructure } from "@/lib/api";
import MarkdownContent from "./MarkdownContent";

const STAGE_LABELS: Record<string, string> = {
  complete: "Complete Report",
  analysts: "Analysts",
  research: "Research",
  trading: "Trading",
  risk: "Risk",
  portfolio: "Portfolio",
};

const STAGE_ORDER = ["complete", "analysts", "research", "trading", "risk", "portfolio"];

const FILE_LABELS: Record<string, string> = {
  market: "Market",
  sentiment: "Sentiment",
  news: "News",
  fundamentals: "Fundamentals",
  bull: "Bull",
  bear: "Bear",
  manager: "Manager",
  trader: "Trader",
  aggressive: "Aggressive",
  conservative: "Conservative",
  neutral: "Neutral",
  decision: "Decision",
};

interface Props {
  reportId: string;
}

export default function ReportViewer({ reportId }: Props) {
  const [structure, setStructure] = useState<ReportStructure | null>(null);
  const [activeStage, setActiveStage] = useState("complete");
  const [activeFile, setActiveFile] = useState("");
  const [content, setContent] = useState("");
  const [loading, setLoading] = useState(false);

  // Load structure when reportId changes
  useEffect(() => {
    getReportStructure(reportId).then((s) => {
      setStructure(s);
      setActiveStage("complete");
      setActiveFile("");
    });
  }, [reportId]);

  // Load content when stage/file changes
  useEffect(() => {
    if (!reportId) return;
    setLoading(true);
    getReportContent(reportId, activeStage, activeFile)
      .then(setContent)
      .finally(() => setLoading(false));
  }, [reportId, activeStage, activeFile]);

  if (!structure) return <p className="text-gray-400">Loading structure…</p>;

  const availableStages = STAGE_ORDER.filter(
    (s) => s === "complete" ? structure.has_complete : structure.stages[s]
  );

  // When switching stage, default to first file in that stage
  const handleStageChange = (stage: string) => {
    setActiveStage(stage);
    if (stage === "complete") {
      setActiveFile("");
    } else {
      const files = structure.stages[stage] || [];
      setActiveFile(files[0] || "");
    }
  };

  const currentFiles = activeStage !== "complete" ? (structure.stages[activeStage] || []) : [];

  return (
    <div className="flex flex-col gap-4">
      {/* Report ID header */}
      <h2 className="text-xl font-semibold text-gray-800">
        {reportId.replace(/_/g, " • ")}
      </h2>

      {/* Stage tabs */}
      <div className="flex gap-1 border-b border-gray-200">
        {availableStages.map((stage) => (
          <button
            key={stage}
            onClick={() => handleStageChange(stage)}
            className={`px-4 py-2 text-sm font-medium transition-colors border-b-2 -mb-px ${
              activeStage === stage
                ? "border-blue-500 text-blue-600"
                : "border-transparent text-gray-500 hover:text-gray-700"
            }`}
          >
            {STAGE_LABELS[stage]}
          </button>
        ))}
      </div>

      {/* File sub-tabs (for multi-file stages) */}
      {currentFiles.length > 1 && (
        <div className="flex gap-1">
          {currentFiles.map((file) => (
            <button
              key={file}
              onClick={() => setActiveFile(file)}
              className={`px-3 py-1 text-xs rounded-full transition-colors ${
                activeFile === file
                  ? "bg-blue-100 text-blue-700 font-medium"
                  : "bg-gray-100 text-gray-500 hover:bg-gray-200"
              }`}
            >
              {FILE_LABELS[file] || file}
            </button>
          ))}
        </div>
      )}

      {/* Content */}
      <div className="bg-white rounded-lg border border-gray-200 p-6">
        {loading ? (
          <p className="text-gray-400 text-sm">Loading…</p>
        ) : (
          <MarkdownContent content={content} />
        )}
      </div>
    </div>
  );
}
```

const API_BASE = process.env.NEXT_PUBLIC_API_URL || "http://localhost:8000";

interface Props {
  onComplete: (reportId: string) => void;
}

export default function AnalyzeForm({ onComplete }: Props) {
  const [ticker, setTicker] = useState("");
  const [date, setDate] = useState(new Date().toISOString().split("T")[0]);
  const [running, setRunning] = useState(false);
  const [messages, setMessages] = useState<string[]>([]);
  const [error, setError] = useState("");

  const addMessage = (msg: string) =>
    setMessages((prev) => [...prev, msg]);

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault();
    if (!ticker.trim()) return;

    setRunning(true);
    setMessages([]);
    setError("");

    try {
      const { job_id } = await startAnalysis({
        ticker: ticker.trim().toUpperCase(),
        date,
      });

      addMessage(`Job started: ${job_id}`);

      // Connect to SSE stream
      const es = new EventSource(
        `${API_BASE}/api/analyze/stream?job_id=${job_id}`
      );

      es.addEventListener("progress", (e) => addMessage(e.data));
      es.addEventListener("ping", () => {}); // ignore keepalive
      es.addEventListener("done", (e) => {
        addMessage(`✓ ${e.data}`);
      });
      es.addEventListener("report_id", (e) => {
        es.close();
        setRunning(false);
        onComplete(e.data);
      });
      es.addEventListener("error", (e: MessageEvent) => {
        setError(e.data || "Analysis failed");
        es.close();
        setRunning(false);
      });

      es.onerror = () => {
        if (running) {
          setError("Connection to server lost");
          es.close();
          setRunning(false);
        }
      };
    } catch (err) {
      setError(String(err));
      setRunning(false);
    }
  };

  return (
    <div className="bg-white border border-gray-200 rounded-lg p-6 max-w-lg">
      <h3 className="font-semibold text-gray-800 mb-4">Run New Analysis</h3>

      <form onSubmit={handleSubmit} className="flex flex-col gap-4">
        <div className="flex gap-3">
          <div className="flex-1">
            <label className="text-xs text-gray-500 mb-1 block">Ticker</label>
            <input
              type="text"
              value={ticker}
              onChange={(e) => setTicker(e.target.value.toUpperCase())}
              placeholder="SPY"
              disabled={running}
              className="w-full border border-gray-300 rounded px-3 py-2 text-sm focus:outline-none focus:ring-2 focus:ring-blue-300 disabled:bg-gray-50"
            />
          </div>
          <div className="flex-1">
            <label className="text-xs text-gray-500 mb-1 block">Date</label>
            <input
              type="date"
              value={date}
              onChange={(e) => setDate(e.target.value)}
              disabled={running}
              className="w-full border border-gray-300 rounded px-3 py-2 text-sm focus:outline-none focus:ring-2 focus:ring-blue-300 disabled:bg-gray-50"
            />
          </div>
        </div>

        <button
          type="submit"
          disabled={running || !ticker.trim()}
          className="px-4 py-2 bg-blue-600 text-white text-sm rounded hover:bg-blue-700 disabled:opacity-50 disabled:cursor-not-allowed transition-colors"
        >
          {running ? "Analyzing…" : "Run Analysis"}
        </button>
      </form>

      {/* Progress log */}
      {messages.length > 0 && (
        <div className="mt-4 bg-gray-50 rounded p-3 max-h-40 overflow-y-auto">
          {messages.map((msg, i) => (
            <p key={i} className="text-xs text-gray-600 font-mono">{msg}</p>
          ))}
        </div>
      )}

      {error && (
        <p className="mt-2 text-xs text-red-600">{error}</p>
      )}
    </div>
  );
}
```
